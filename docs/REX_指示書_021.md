## 設計概要

### 戦略の本質（確認）
「4H上昇ダウが継続している限り、押し目条件が揃うたびにエントリーを繰り返す」

### 新アーキテクチャ（トップダウン3層スキャン）
```
Layer 1: 4H 構造スキャン
  → 4H LONG 期間を特定 + 各 4H SL のタイムスタンプを取得

Layer 2: 1H 窓確定
  → 4H SL ts から 1H 窓（前20本 + SL足 + 後5本 = 26本）を確定

Layer 3: 窓内 15M/5M スキャン
  → 窓内 5M → 15M にリサンプル
  → check_15m_range_low() で DB/IHS 判定
  → 15M ネック越え 5M 実体確定 → エントリー
```

### エントリー条件（シンプル版 — Phase 1）
```
前提:
  - get_direction_4h() == 'LONG'
  - 4H SL タイムスタンプ ±8本 窓内に 1H SL 存在（#020確認済み: 100%）

エントリー:
  ① 窓内 15M で DB or IHS パターン成立（check_15m_range_low）
  ② 窓内 5M 足の min(open,close) > 15M neck（実体上抜け確定）
  ③ 確定足の次の 5M 足の始値でエントリー

※ Phase 2（将来）: ①を「15M 右肩内で 5M DB ネック越え」に拡張
```

### 決済ロジック（既存 exit_logic.py をそのまま使用）
```
初動SL : 15M ダウ崩れ → 次足始値で全量損切
段階1  : 5M Swing確定後 → 5M ダウ崩れで全量決済
段階2  : 4H ネック到達 → 半値決済 + 建値SL移動
段階3  : 1H 実体確定後 → 15M ダウ崩れで残量決済
```

```python
"""
REX Window Scanner — 窓ベース階層スキャン（Phase 1: シンプル版）

4H上昇トレンド → 1H窓確定 → 窓内15M DB/IHS → 5Mネック越えエントリー
既存 backtest.py を変更せず、独立したスキャナーとして動作する。
"""
import pandas as pd
import numpy as np
import sys
from pathlib import Path

sys.path.insert(0, str(Path(__file__).parent))

from swing_detector import (
    detect_swing_highs,
    detect_swing_lows,
    get_direction_4h,
)
from entry_logic import check_15m_range_low, PIP_SIZE
from exit_logic import manage_exit

# ========== 定数 ==========
DATA_PATH = Path(__file__).parent.parent / 'data/raw/usdjpy_multi_tf_2years.parquet'
PIP = 0.01
WARMUP_4H = 50
WINDOW_1H_PRE  = 20   # 1H SL 足の前 20本
WINDOW_1H_POST = 5    # 1H SL 足の後 5本
WINDOW_SEARCH  = 8    # 4H SL ts 前後 ±8本で 1H SL を探す範囲
DIRECTION_MODE = 'LONG'  # SHORT は一時停止中

# ========== リサンプル（backtest.py と同一。変更禁止）==========
def resample_tf(df, rule):
    return df.resample(rule, label='right', closed='right').agg(
        {'open': 'first', 'high': 'max', 'low': 'min', 'close': 'last'}
    ).dropna()


# ========== Layer 1: 4H 構造スキャン ==========
def scan_4h_events(df_4h):
    """
    4H LONG 期間の各バーについて、
    4H SL のタイムスタンプと価格を返すジェネレーター。
    
    Yields: (i, ts_4h, sl_4h_val, sl_4h_ts)
    """
    for i in range(WARMUP_4H, len(df_4h)):
        direction = get_direction_4h(
            df_4h['high'], df_4h['low'],
            current_idx=i, n=3, lookback=20
        )
        if direction != DIRECTION_MODE:
            continue

        # 4H SL のタイムスタンプと価格を取得
        start = max(0, i - 20 + 1)
        window = df_4h['low'].iloc[start:i+1]
        mask = detect_swing_lows(window, n=3)
        sl_series = window[mask]
        if len(sl_series) == 0:
            continue

        yield (i, df_4h.index[i], float(sl_series.iloc[-1]), sl_series.index[-1])


# ========== Layer 2: 1H 窓確定 ==========
def get_1h_window_range(df_1h, sl_4h_ts):
    """
    4H SL タイムスタンプを基点に 1H 窓の開始/終了タイムスタンプを返す。
    
    Returns: (win_start_ts, win_end_ts, sl_1h_ts) or None
    """
    idx_center = df_1h.index.searchsorted(sl_4h_ts, side='right') - 1
    if idx_center < WINDOW_SEARCH:
        return None

    # ±WINDOW_SEARCH 本で 1H SL を探す
    ws = max(0, idx_center - WINDOW_SEARCH)
    we = min(len(df_1h) - 1, idx_center + WINDOW_SEARCH)
    search_window = df_1h['low'].iloc[ws:we+1]
    mask = detect_swing_lows(search_window, n=2)
    sl_1h_near = search_window[mask]

    if len(sl_1h_near) == 0:
        return None

    # 最も 4H SL に価格が近い 1H SL を選択
    sl_4h_val = df_1h['low'].iloc[idx_center]  # 近似値として使用
    dists = abs(sl_1h_near - sl_4h_val)
    sl_1h_ts = dists.idxmin()

    # 窓範囲を確定
    idx_sl = df_1h.index.get_loc(sl_1h_ts)
    win_start = df_1h.index[max(0, idx_sl - WINDOW_1H_PRE)]
    win_end   = df_1h.index[min(len(df_1h) - 1, idx_sl + WINDOW_1H_POST)]

    return (win_start, win_end, sl_1h_ts)


# ========== Layer 3: 窓内スキャン ==========
def scan_window_entry(df_5m_win, sl_4h_val):
    """
    窓内の 5M データから:
      1. 15M にリサンプル
      2. check_15m_range_low() で DB/IHS 判定
      3. ネック越え 5M 実体確定を検出
    
    Returns: dict with entry info, or None
    """
    if len(df_5m_win) < 30:  # 最低30本（2.5時間）は必要
        return None

    # 窓内 5M → 15M リサンプル
    df_15m_win = resample_tf(df_5m_win, '15min')
    if len(df_15m_win) < 10:
        return None

    # 15M SH/SL 検出
    sh_15m = detect_swing_highs(df_15m_win['high'], n=3)
    sl_15m = detect_swing_lows(df_15m_win['low'], n=3)

    sh_vals = df_15m_win['high'][sh_15m]
    sl_vals = df_15m_win['low'][sl_15m]

    if len(sl_vals) < 2 or len(sh_vals) < 1:
        return None  # DB/IHS には最低 SL 2本 + SH 1本 が必要

    # ---- check_15m_range_low() を呼ぶ ----
    # この関数は (pattern, neck_15m, details) を返す
    # 引数: 15M の SH/SL 系列を渡す
    # ※ 既存関数のシグネチャを確認してから呼ぶこと。
    #    もしシグネチャが異なる場合は、Rex に報告して停止。
    #
    # 期待するシグネチャ:
    #   check_15m_range_low(sl_15m_series, sh_15m_series, ...) 
    #   → (pattern: str|None, neck: float|None, details: dict)
    #
    # ※※※ 重要 ※※※
    # ClaudeCode へ: 必ず entry_logic.py の check_15m_range_low() の
    # 実際のシグネチャを read してから、適切な引数で呼び出すこと。
    # 指示書のコードをそのままコピペせず、実際のAPIに合わせること。
    # #020 で get_direction_4h の current_idx 問題があったのと同じ。

    try:
        result = check_15m_range_low(
            low_15m=df_15m_win['low'],
            high_15m=df_15m_win['high'],
        )
    except TypeError as e:
        print(f"  !! check_15m_range_low API mismatch: {e}")
        print(f"  !! Read entry_logic.py and fix the call signature.")
        return None

    if result is None:
        return None

    pattern, neck_15m = result[0], result[1]
    if pattern is None or neck_15m is None:
        return None

    # ---- 5M ネック越え実体確定を検出 ----
    for j in range(len(df_5m_win) - 1):
        row = df_5m_win.iloc[j]
        body_low = min(row['open'], row['close'])

        if body_low > neck_15m:
            # 確定足 = j, 執行足 = j+1
            entry_bar = df_5m_win.iloc[j + 1]
            return {
                'pattern':     pattern,
                'neck_15m':    neck_15m,
                'confirm_ts':  df_5m_win.index[j],
                'entry_ts':    df_5m_win.index[j + 1],
                'entry_price': float(entry_bar['open']),
                'sl_4h':       sl_4h_val,
            }

    return None  # ネック越え未発生


# ========== メイン実行 ==========
def run_window_scan():
    """窓ベース階層スキャンのメインループ。"""

    # ---- データ読み込み ----
    df_raw = pd.read_parquet(DATA_PATH)
    df_5m = df_raw[["5M_Open", "5M_High", "5M_Low", "5M_Close"]].rename(
        columns={"5M_Open": "open", "5M_High": "high",
                 "5M_Low": "low", "5M_Close": "close"}
    )
    df_4h = resample_tf(df_5m, '4h')
    df_1h = resample_tf(df_5m, '1h')

    print(f"Data: 5M={len(df_5m)} / 1H={len(df_1h)} / 4H={len(df_4h)}")

    # ---- スキャン ----
    entries = []
    seen_windows = set()  # 同一窓の重複スキップ用

    for (i_4h, ts_4h, sl_4h_val, sl_4h_ts) in scan_4h_events(df_4h):

        # 同一 4H SL タイムスタンプの窓は1回だけ処理
        window_key = sl_4h_ts
        if window_key in seen_windows:
            continue
        seen_windows.add(window_key)

        # Layer 2: 1H 窓確定
        win_result = get_1h_window_range(df_1h, sl_4h_ts)
        if win_result is None:
            continue
        win_start, win_end, sl_1h_ts = win_result

        # 窓内 5M データ抽出
        df_5m_win = df_5m.loc[win_start:win_end]
        if len(df_5m_win) < 30:
            continue

        # Layer 3: 窓内エントリースキャン
        entry = scan_window_entry(df_5m_win, sl_4h_val)
        if entry is None:
            continue

        entry['ts_4h'] = ts_4h
        entry['sl_4h_ts'] = sl_4h_ts
        entry['sl_1h_ts'] = sl_1h_ts
        entry['window'] = f"{win_start} ~ {win_end}"
        entries.append(entry)

        print(f"  ENTRY: {entry['entry_ts']} @ {entry['entry_price']:.3f} "
              f"| pat={entry['pattern']} | neck={entry['neck_15m']:.3f}")

    # ---- 結果出力 ----
    df_entries = pd.DataFrame(entries)
    total = len(df_entries)
    print(f"\n=== Window Scanner Phase 1 結果 ===")
    print(f"スキャン窓数     : {len(seen_windows)}")
    print(f"エントリー検出数 : {total}")

    if total > 0:
        # パターン別集計
        for pat in ['DB', 'IHS', 'ASCENDING']:
            subset = df_entries[df_entries['pattern'] == pat]
            print(f"  {pat}: {len(subset)} 件")

    # CSV 保存
    out_path = Path(__file__).parent.parent / 'logs/window_scan_entries.csv'
    out_path.parent.mkdir(parents=True, exist_ok=True)
    df_entries.to_csv(out_path, index=False)
    print(f"\n結果CSV: {out_path}")

    return df_entries


if __name__ == '__main__':
    run_window_scan()
```
