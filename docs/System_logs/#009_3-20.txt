    if len(sl_prices) < 2:
        return {'found': False}

    sl_min  = float(sl_prices.min())      # 最深値（頭）

    sl_last = float(sl_prices.iloc[-1])   # 最終安値（右肩）

### 1-2. 新規追加: check_15m_range_low()

```python
def check_15m_range_low(low_15m: pd.Series,
                         high_15m: pd.Series,
                         direction: str,
                         n: int = 3,
                         lookback: int = 40) -> dict:
    """
    統合レンジロジック — ダブルボトム・逆三尊・安値切り上げを統合検出。

    LONG 成立条件:
      ① SL_last >= SL_min
           最終安値（右肩）が最深値（頭）を下回らない

      ② SL_last <= SL_min + 2.0 * (SL2 - SL_min)
           右肩の上限 = SL_min から SL2 の2倍の値幅まで
           （画像確認済みの等距離ルール）

      ③ SL_min 以降に 15M Swing High が存在する
           ネックライン形成確認

    ネックライン定義:
      SL_min（最深値）のインデックス以降の
      15M Swing High の最高値

    SHORT は High 系列で対称に実装すること。

    戻り値:
      {
        'found'    : True / False,
        'sl_min'   : float,   # 最深値（頭・1番底）
        'sl_last'  : float,   # 最終安値（右肩）
        'neck_15m' : float,   # ネックライン価格
        'pattern'  : str,     # 'DB' / 'IHS' / 'ASCENDING'（デバッグ用）
        'reason'   : str,     # found=False の時の理由
      }
    """
    from swing_detector import detect_swing_lows, detect_swing_highs

    if direction == 'LONG':
        window_low  = low_15m.iloc[-lookback:]
        window_high = high_15m.iloc[-lookback:]

        sl_mask   = detect_swing_lows(window_low, n=n)
        sl_prices = window_low[sl_mask]

        if len(sl_prices) < 2:
            return {'found': False, 'reason': '15M SL 不足（2本未満）'}

        sl_min   = float(sl_prices.min())
        sl2      = float(sl_prices.iloc[-2])
        sl_last  = float(sl_prices.iloc[-1])

        # 条件①: 最終安値 >= 最深値
        if sl_last < sl_min:
            return {'found': False, 'reason': 'SL_last < SL_min（最深値割れ）'}

        # 条件②: 右肩の上限チェック
        sl3_max = sl_min + 2.0 * (sl2 - sl_min)
        if sl_last > sl3_max:
            return {'found': False, 'reason': 'SL3 上限超過（等距離ルール違反）'}

        # 条件③: ネックライン取得
        sl_min_idx      = sl_prices.idxmin()
        neck_candidates = window_high[window_high.index > sl_min_idx]
        sh_mask         = detect_swing_highs(neck_candidates, n=n)
        sh_prices       = neck_candidates[sh_mask]

        if len(sh_prices) == 0:
            return {'found': False, 'reason': 'SL_min 以降に 15M SH なし'}

        neck_15m = float(sh_prices.max())

        # パターン分類（デバッグ・統計用）
        threshold = 0.03 * abs(sl2 - sl_min)
        if abs(sl_last - sl2) <= threshold:
            pattern = 'DB'
        elif sl_last <= sl2:
            pattern = 'IHS'
        else:
            pattern = 'ASCENDING'

        return {
            'found'    : True,
            'sl_min'   : sl_min,
            'sl_last'  : sl_last,
            'neck_15m' : neck_15m,
            'pattern'  : pattern,
            'reason'   : 'OK',
        }

    elif direction == 'SHORT':
        # HIGH 系列で対称に実装
        # SH_last <= SH_max（最終高値が最高値を上回らない）
        # SH_last >= SH_max - 2.0 * (SH_max - SH2)（下限チェック）
        # ネックライン = SH_max 以降の 15M Swing Low の最安値

        window_high = high_15m.iloc[-lookback:]
        window_low  = low_15m.iloc[-lookback:]

        sh_mask   = detect_swing_highs(window_high, n=n)
        sh_prices = window_high[sh_mask]

        if len(sh_prices) < 2:
            return {'found': False, 'reason': '15M SH 不足（2本未満）'}

        sh_max   = float(sh_prices.max())
        sh2      = float(sh_prices.iloc[-2])
        sh_last  = float(sh_prices.iloc[-1])

        if sh_last > sh_max:
            return {'found': False, 'reason': 'SH_last > SH_max（最高値超え）'}

        sh3_min = sh_max - 2.0 * (sh_max - sh2)
        if sh_last < sh3_min:
            return {'found': False, 'reason': 'SH3 下限超過（等距離ルール違反）'}

        sh_max_idx      = sh_prices.idxmax()
        neck_candidates = window_low[window_low.index > sh_max_idx]
        sl_mask         = detect_swing_lows(neck_candidates, n=n)
        sl_prices       = neck_candidates[sl_mask]

        if len(sl_prices) == 0:
            return {'found': False, 'reason': 'SH_max 以降に 15M SL なし'}

        neck_15m = float(sl_prices.min())

        threshold = 0.03 * abs(sh2 - sh_max)
        if abs(sh_last - sh2) <= threshold:
            pattern = 'DT'
        elif sh_last >= sh2:
            pattern = 'HNS'
        else:
            pattern = 'DESCENDING'

        return {
            'found'    : True,
            'sl_min'   : sh_max,
            'sl_last'  : sh_last,
            'neck_15m' : neck_15m,
            'pattern'  : pattern,
            'reason'   : 'OK',
        }

    return {'found': False, 'reason': 'direction 不明'}
```

## Rex から ClaudeCode への補足

### SHORT 版の変数名について

SHORT 版の戻り値 `sl_min` キーに `sh_max`（最高値）を格納している。
呼び出し側で混乱しないよう、backtest.py では以下の読み替えを行うこと:

```python
result = check_15m_range_low(...)
if result['found']:
    neck_15m = result['neck_15m']   # LONG/SHORT 共通キーなのでそのまま使える
    # sl_min キーは LONG=最深安値 / SHORT=最高値 なので直接参照しないこと
```

### SL3 上限チェックの数値確認

SL3 上限 = SL_min + 2.0 × (SL2 - SL_min) の計算で
SL2 < SL_min のケース（DB では SL2 = SL1 が SL_min と同水準）は
threshold がほぼゼロになる。

この場合 sl3_max ≒ sl_min となり条件②が即 False になるバグが出る可能性がある。
実装時に以下を追加すること:

```python
# SL2 が SL_min と同水準（DB パターン）の場合の保護
MIN_RANGE = 5.0 * PIP_SIZE   # 最低 5pips の上限余裕を確保
sl3_max = max(sl_min + 2.0 * (sl2 - sl_min), sl_min + MIN_RANGE)
```

### パターン分類の閾値

```python
threshold = 0.03 * abs(sl2 - sl_min)
```

この 3% 閾値は「SL2 と SL_last がほぼ同水準 = DB」の判定に使う。
USDJPY 5M の値動きで 3% は約 0.1〜0.2 pips 相当。
厳しすぎる場合は 5% に変更してよい（デバッグ出力で比率確認後に調整）。
