📐 Swing High / Low 検出ロジック — 詳細設計
核心の考え方（超シンプル）
「あるローソク足の高値が、前後N本の中で一番高いならSwing High」— これだけだよ。現在の prev_high（1本前のHigh）との比較とは全く別物で、構造的なピーク・谷を検出することができる。

Pythonコード — swing_detector.py（新規作成）
python
# swing_detector.py
# ミナト流 Swing High / Low 検出モジュール
# Phase A 最優先実装

import pandas as pd
import numpy as np

def detect_swing_highs(high: pd.Series, n: int = 3) -> pd.Series:
    """
    Swing High（局所的な高値ピーク）を検出する
    
    考え方：
      足[i]のHighが、前後n本のHighより全部高ければ → Swing High確定
    
    引数:
      high : 各足のHighのSeries（4H/1H/15M/5M どれでもOK）
      n    : 左右に何本確認するか（4H=5, 1H=3, 15M=3, 5M=2 が目安）
    
    戻り値:
      True/False のSeries（Trueの足がSwing High）
    """
    swing_high = pd.Series(False, index=high.index)
    
    for i in range(n, len(high) - n):
        # 左側n本 と 右側n本 が全部今の足より低いか確認
        left_ok  = all(high.iloc[i] > high.iloc[i - j] for j in range(1, n + 1))
        right_ok = all(high.iloc[i] > high.iloc[i + j] for j in range(1, n + 1))
        
        if left_ok and right_ok:
            swing_high.iloc[i] = True
    
    return swing_high


def detect_swing_lows(low: pd.Series, n: int = 3) -> pd.Series:
    """
    Swing Low（局所的な安値の谷）を検出する
    
    引数・戻り値はdetect_swing_highsと対称
    """
    swing_low = pd.Series(False, index=low.index)
    
    for i in range(n, len(low) - n):
        left_ok  = all(low.iloc[i] < low.iloc[i - j] for j in range(1, n + 1))
        right_ok = all(low.iloc[i] < low.iloc[i + j] for j in range(1, n + 1))
        
        if left_ok and right_ok:
            swing_low.iloc[i] = True
    
    return swing_low


def get_nearest_swing_high(high: pd.Series, current_idx: int, n: int = 3,
                           lookback: int = 50) -> float:
    """
    現在の足より前の、直近のSwing Highの値を返す
    
    使い方：ストップライン、ネックラインの基準価格として使う
    
    引数:
      current_idx : 今見ている足の位置（整数インデックス）
      lookback    : 何本前まで遡るか（4H=20, 1H=50 など）
    """
    window = high.iloc[max(0, current_idx - lookback): current_idx]
    swings = detect_swing_highs(window, n=n)
    
    swing_prices = window[swings]  # Swing HighのHighだけ抜き出す
    
    if len(swing_prices) == 0:
        # 見つからない場合は直近の最高値で代用
        return window.max()
    
    # 直近（一番右）のSwing Highを返す
    return float(swing_prices.iloc[-1])


def get_nearest_swing_low(low: pd.Series, current_idx: int, n: int = 3,
                          lookback: int = 50) -> float:
    """
    現在の足より前の、直近のSwing Lowの値を返す
    
    使い方：4H押し目の基準、ネックライン（半値利確ターゲット）として使う
    """
    window = low.iloc[max(0, current_idx - lookback): current_idx]
    swings = detect_swing_lows(window, n=n)
    
    swing_prices = window[swings]
    
    if len(swing_prices) == 0:
        return window.min()
    
    return float(swing_prices.iloc[-1])

def get_direction(df_4h: pd.DataFrame, idx: int) -> str:
    """
    4H足のSwing構造から方向を判定する
    戻り値: 'LONG' or 'SHORT' or 'NONE'（判定不能）
    """
    sh = get_nearest_swing_high(df_4h['High'], idx, n=5, lookback=20)
    sl = get_nearest_swing_low(df_4h['Low'],  idx, n=5, lookback=20)

    # 上昇ダウ: 高値・安値ともに切り上がっているか
    prev_sh = get_nearest_swing_high(df_4h['High'], idx - 5, n=5, lookback=20)
    prev_sl = get_nearest_swing_low(df_4h['Low'],   idx - 5, n=5, lookback=20)

    if sh > prev_sh and sl > prev_sl:
        return 'LONG'   # 高値・安値ともに切り上がり
    elif sh < prev_sh and sl < prev_sl:
        return 'SHORT'  # 高値・安値ともに切り下がり
    else:
        return 'NONE'   # トレンドレス → ノーポジ

# entry_logic.py Step1 の実装イメージ
def check_fib_condition(price, swing_high, swing_low, neck_4h):
    """
    戻り値: 優位性スコア（0=条件外, 1=基本, 2=高優位性）
    """
    retracement = (swing_high - price) / (swing_high - swing_low)

    at_fib50   = 0.45 <= retracement <= 0.55   # Fib50%±5%
    at_fib618  = retracement <= 0.65            # Fib61.8%以内
    near_neck  = abs(price - neck_4h) < (swing_high - swing_low) * 0.03  # ネック付近3%以内

    if at_fib50 and near_neck:
        return 2   # ★★★ 最高優位性（Fib50% + ネックライン一致）
    elif at_fib618:
        return 1   # ★★  基本条件（Fib61.8%以内）
    else:
        return 0   # 条件外

### Step1: 4H押し目確認（Fib条件）

def check_fib_condition(price: float,
                         swing_high: float,
                         swing_low: float,
                         neck_4h: float,
                         direction: str) -> int:
    """
    Fibonacci押し目条件を確認する。

    戻り値:
      2 = 高優位性（Fib50%付近 かつ 4Hネックライン付近）
      1 = 基本条件（Fib61.8%以内）
      0 = 条件外

    direction='LONG'  の場合: 押し目（retracement = 高値からの戻り率）
    direction='SHORT' の場合: 戻り（retracement = 安値からの戻り率・逆方向）
    """
    if direction == 'LONG':
        if swing_high == swing_low:
            return 0
        retracement = (swing_high - price) / (swing_high - swing_low)
        near_neck = abs(price - neck_4h) < (swing_high - swing_low) * 0.03
    elif direction == 'SHORT':
        if swing_high == swing_low:
            return 0
        retracement = (price - swing_low) / (swing_high - swing_low)
        near_neck = abs(price - neck_4h) < (swing_high - swing_low) * 0.03
    else:
        return 0

    at_fib50  = 0.45 <= retracement <= 0.55
    at_fib618 = retracement <= 0.65

    if at_fib50 and near_neck:
        return 2   # ★★★ 最高優位性
    elif at_fib618:
        return 1   # ★★ 基本条件
    else:
        return 0   # 条件外


### Step2: 1H 2番底確認

def check_double_bottom_1h(low_1h: pd.Series,
                            current_idx: int,
                            direction: str,
                            n: int = 3,
                            lookback: int = 50) -> bool:
    """
    1H足での2番底（direction=LONG）または2番天井（direction=SHORT）を確認する。

    LONG の場合:
      直近SL①を確認 → 反発上昇 → SL②が形成され SL② >= SL① なら成立
      （安値が切り上がっている = 2番底成立）

    SHORT の場合:
      直近SH①を確認 → 反落 → SH②が形成され SH② <= SH① なら成立
      （高値が切り下がっている = 2番天井成立）

    戻り値: True / False
    """
    from swing_detector import detect_swing_lows, detect_swing_highs

    window = low_1h.iloc[max(0, current_idx - lookback): current_idx + 1]

    if direction == 'LONG':
        swings = detect_swing_lows(window, n=n)
        swing_prices = window[swings]
        if len(swing_prices) < 2:
            return False
        sl1 = float(swing_prices.iloc[-2])
        sl2 = float(swing_prices.iloc[-1])
        return sl2 >= sl1   # 安値切り上がり = 2番底成立

    elif direction == 'SHORT':
        from swing_detector import detect_swing_highs
        high_1h_window = low_1h.iloc[
            max(0, current_idx - lookback): current_idx + 1
        ]
        # ※ SHORT用は high_1h を引数で受け取る必要あり（下記Step2呼び出し参照）
        return False  # SHORT用は呼び出し側で high_1h を渡す形に調整すること

    return False


def check_double_top_1h(high_1h: pd.Series,
                         current_idx: int,
                         n: int = 3,
                         lookback: int = 50) -> bool:
    """
    SHORT用: 1H足での2番天井確認
    直近SH① → 反落 → SH② <= SH① なら成立
    """
    from swing_detector import detect_swing_highs
    window = high_1h.iloc[max(0, current_idx - lookback): current_idx + 1]
    swings = detect_swing_highs(window, n=n)
    swing_prices = window[swings]
    if len(swing_prices) < 2:
        return False
    sh1 = float(swing_prices.iloc[-2])
    sh2 = float(swing_prices.iloc[-1])
    return sh2 <= sh1   # 高値切り下がり = 2番天井成立


### Step3: 15Mネックライン越え（5M実体確定足）

def check_15m_neckline_break(close_5m: pd.Series,
                              open_5m: pd.Series,
                              neck_15m: float,
                              direction: str) -> bool:
    """
    15Mネックラインを5M実体が越えたかどうかを確認する（確定足判定）。

    確定足の定義:
      LONG : 5M実体（min(open, close)）がneck_15mを上回っている
      SHORT: 5M実体（max(open, close)）がneck_15mを下回っている

    現在バー（最新の5M足）に対して判定する。
    """
    latest_close = float(close_5m.iloc[-1])
    latest_open  = float(open_5m.iloc[-1])
    body_low  = min(latest_open, latest_close)
    body_high = max(latest_open, latest_close)

    if direction == 'LONG':
        return body_low > neck_15m    # 実体全体がネック上 = 確定
    elif direction == 'SHORT':
        return body_high < neck_15m   # 実体全体がネック下 = 確定
    return False


### 統合エントリー判定関数

def evaluate_entry(price: float,
                   direction: str,
                   swing_high_4h: float,
                   swing_low_4h: float,
                   neck_4h: float,
                   low_1h: pd.Series,
                   high_1h: pd.Series,
                   current_idx_1h: int,
                   close_5m: pd.Series,
                   open_5m: pd.Series,
                   neck_15m: float) -> dict:
    """
    3段階エントリー条件を一括評価する。

    戻り値:
      {
        'enter'    : bool   # エントリー可否
        'fib_score': int    # Fib優位性スコア（0/1/2）
        'reason'   : str    # スキップ理由（デバッグ用）
      }
    """
    # Step1: Fib条件
    fib_score = check_fib_condition(
        price, swing_high_4h, swing_low_4h, neck_4h, direction
    )
    if fib_score == 0:
        return {'enter': False, 'fib_score': 0, 'reason': 'Fib条件外'}

    # Step2: 1H 2番底/2番天井
    if direction == 'LONG':
        db_ok = check_double_bottom_1h(low_1h, current_idx_1h, direction)
    else:
        db_ok = check_double_top_1h(high_1h, current_idx_1h)

    if not db_ok:
        return {'enter': False, 'fib_score': fib_score, 'reason': '1H2番底未形成'}

    # Step3: 15Mネック越え（5M実体確定）
    neck_ok = check_15m_neckline_break(close_5m, open_5m, neck_15m, direction)
    if not neck_ok:
        return {'enter': False, 'fib_score': fib_score, 'reason': '15Mネック未越え'}

    return {'enter': True, 'fib_score': fib_score, 'reason': 'OK'}

#### 1. check_5m_dow_break（5Mダウ崩れ判定）

def check_5m_dow_break(close_5m: pd.Series,
                        open_5m: pd.Series,
                        direction: str,
                        n: int = 2) -> bool:
    """
    5M足のSwing押し戻しラインを5M実体が越えたか判定する。
    （確定足判定 = 実体がラインを越えた足）

    LONG の場合：
      直近5M Swing Lowを割り込む5M実体 → True（ダウ崩れ）
      body_high = min(open, close) がSwingLowを下回る

    SHORT の場合：
      直近5M Swing Highを上抜ける5M実体 → True（ダウ崩れ）
      body_low = max(open, close) がSwingHighを上回る

    戻り値: True = 崩れ確定（次の5M始値で決済執行）
    """
    from swing_detector import get_nearest_swing_low, get_nearest_swing_high

    current_idx = len(close_5m) - 1
    latest_close = float(close_5m.iloc[-1])
    latest_open  = float(open_5m.iloc[-1])
    body_low  = min(latest_open, latest_close)
    body_high = max(latest_open, latest_close)

    if direction == 'LONG':
        swing_low = get_nearest_swing_low(
            close_5m, current_idx, n=n, lookback=20
        )
        return body_high < swing_low  # 実体全体がSwingLow下 = 崩れ確定

    elif direction == 'SHORT':
        swing_high = get_nearest_swing_high(
            close_5m, current_idx, n=n, lookback=20
        )
        return body_low > swing_high  # 実体全体がSwingHigh上 = 崩れ確定

    return False


#### 2. check_15m_dow_break（15Mダウ崩れ判定）

def check_15m_dow_break(close_15m: pd.Series,
                         open_15m: pd.Series,
                         direction: str,
                         n: int = 3) -> bool:
    """
    15M足のSwing押し戻しラインを15M実体が越えたか判定する。
    4Hネックライン越え後の最終決済用。

    ロジックは check_5m_dow_break と同じ。TFが15Mになる点のみ異なる。
    """
    from swing_detector import get_nearest_swing_low, get_nearest_swing_high

    current_idx = len(close_15m) - 1
    latest_close = float(close_15m.iloc[-1])
    latest_open  = float(open_15m.iloc[-1])
    body_low  = min(latest_open, latest_close)
    body_high = max(latest_open, latest_close)

    if direction == 'LONG':
        swing_low = get_nearest_swing_low(
            close_15m, current_idx, n=n, lookback=30
        )
        return body_high < swing_low

    elif direction == 'SHORT':
        swing_high = get_nearest_swing_high(
            close_15m, current_idx, n=n, lookback=30
        )
        return body_low > swing_high

    return False


#### 3. manage_exit（統合決済管理）

def manage_exit(entry_price: float,
                direction: str,
                close_5m: pd.Series,
                open_5m: pd.Series,
                close_15m: pd.Series,
                open_15m: pd.Series,
                neck_4h: float,
                position_size: float = 1.0,
                reentry_count: int = 0) -> dict:
    """
    ミナト流3段階決済を管理する。

    決済フロー:
      【段階1: 4Hネック未到達】
        5Mダウ崩れ → 全量決済（利確・損切 共通）

      【段階2: 4Hネック到達】
        半値決済（50%クローズ）
        残り50%のストップを建値に移動（ノーリスク化）

      【段階3: 4Hネック越え後】
        15Mダウ崩れ → 残り全量決済

    戻り値:
      {
        'action'        : 'hold' / 'exit_half' / 'exit_all'
        'exit_price_bar': bool  # 次の足の始値で執行するか
        'reason'        : str   # 決済理由
        'remaining_size': float # 残りポジションサイズ
      }
    """
    current_price = float(close_5m.iloc[-1])

    # 4Hネック到達チェック
    if direction == 'LONG':
        at_neck_4h = current_price >= neck_4h
    else:
        at_neck_4h = current_price <= neck_4h

    # 段階1: 4Hネック未到達 → 5Mダウ崩れで全決済
    if not at_neck_4h:
        if check_5m_dow_break(close_5m, open_5m, direction):
            return {
                'action'        : 'exit_all',
                'exit_price_bar': True,   # 次の5M始値で執行
                'reason'        : '5Mダウ崩れ（4Hネック未到達）',
                'remaining_size': 0.0
            }
        return {'action': 'hold', 'exit_price_bar': False,
                'reason': '保有継続', 'remaining_size': position_size}

    # 段階2: 4Hネック到達 → 半値決済
    if position_size == 1.0:  # まだ半値決済していない
        return {
            'action'        : 'exit_half',
            'exit_price_bar': True,   # 次の5M始値で執行
            'reason'        : '4Hネックライン到達・半値決済',
            'remaining_size': 0.5
        }

    # 段階3: 4Hネック越え・残り50%保有中 → 15Mダウ崩れで全決済
    if position_size == 0.5:
        if check_15m_dow_break(close_15m, open_15m, direction):
            return {
                'action'        : 'exit_all',
                'exit_price_bar': True,   # 次の15M始値で執行
                'reason'        : '15Mダウ崩れ（4Hネック越え後・最終決済）',
                'remaining_size': 0.0
            }
        return {'action': 'hold', 'exit_price_bar': False,
                'reason': '4Hネック越え・15Mダウ崩れ待ち', 'remaining_size': 0.5}

    return {'action': 'hold', 'exit_price_bar': False,
            'reason': '保有継続', 'remaining_size': position_size}

### 修正後の関数シグネチャ

def check_5m_dow_break(high_5m: pd.Series,
                        low_5m: pd.Series,
                        close_5m: pd.Series,
                        open_5m: pd.Series,
                        direction: str,
                        n: int = 2) -> bool:
    """
    5M足のSwing押し戻しラインを5M実体が越えたか判定する。

    LONG の場合：
      Low系列 でSwing Low を検出
      body_high = min(open, close)
      body_high < swing_low → True（ダウ崩れ確定）

    SHORT の場合：
      High系列 でSwing High を検出
      body_low = max(open, close)
      body_low > swing_high → True（ダウ崩れ確定）
    """
    from swing_detector import get_nearest_swing_low, get_nearest_swing_high

    current_idx  = len(close_5m) - 1
    latest_close = float(close_5m.iloc[-1])
    latest_open  = float(open_5m.iloc[-1])
    body_low  = min(latest_open, latest_close)
    body_high = max(latest_open, latest_close)

    if direction == 'LONG':
        swing_low = get_nearest_swing_low(
            low_5m, current_idx, n=n, lookback=20   # Low系列を使う
        )
        return body_high < swing_low

    elif direction == 'SHORT':
        swing_high = get_nearest_swing_high(
            high_5m, current_idx, n=n, lookback=20  # High系列を使う
        )
        return body_low > swing_high

    return False


def check_15m_dow_break(high_15m: pd.Series,
                         low_15m: pd.Series,
                         close_15m: pd.Series,
                         open_15m: pd.Series,
                         direction: str,
                         n: int = 3) -> bool:
    """
    check_5m_dow_break と同じ修正を15M足版にも適用する。
    """
    from swing_detector import get_nearest_swing_low, get_nearest_swing_high

    current_idx  = len(close_15m) - 1
    latest_close = float(close_15m.iloc[-1])
    latest_open  = float(open_15m.iloc[-1])
    body_low  = min(latest_open, latest_close)
    body_high = max(latest_open, latest_close)

    if direction == 'LONG':
        swing_low = get_nearest_swing_low(
            low_15m, current_idx, n=n, lookback=30
        )
        return body_high < swing_low

    elif direction == 'SHORT':
        swing_high = get_nearest_swing_high(
            high_15m, current_idx, n=n, lookback=30
        )
        return body_low > swing_high

    return False

## 作業② neck_4h 計算の確認と修正

以下の計算が正しいか確認し、間違っていれば修正すること。

LONG の場合:
  neck_4h = 直近の4H Swing High（上値ターゲット）
  → get_nearest_swing_high(high_4h, idx, n=3) ✅

SHORT の場合:
  neck_4h = 直近の4H Swing Low（下値ターゲット）
  → get_nearest_swing_low(low_4h, idx, n=3)  ← SHORTではLowを使う

現在の実装が両方 get_nearest_swing_high を使っているなら修正すること。
修正箇所: backtest.py の _scan_all_bars_for_entry 内。
